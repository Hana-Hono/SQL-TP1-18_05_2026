--1)
Select titre, pseudo, date_debut_autorisee, heure_debut,
CASE
    WHEN heure_debut >= date_debut_autorisee AND heure_fin <= date_fin_autorisee
    THEN 'VALIDE'
    ELSE 'INVALIDE'
END as validation
	from streamer right join creneau on streamer.id_streamer = creneau.id_streamer
		right join stream on stream.id_creneau = creneau.id_creneau
			order by id_stream

--2)
Select titre, pseudo, date_debut_autorisee, heure_debut,
CASE
    WHEN heure_debut >= date_debut_autorisee AND heure_fin <= date_fin_autorisee
    THEN 'VALIDE'
    ELSE 'INVALIDE'
END as validation
	from streamer right join creneau on streamer.id_streamer = creneau.id_streamer
		right join stream on stream.id_creneau = creneau.id_creneau
			where NOT(heure_debut >= date_debut_autorisee AND heure_fin <= date_fin_autorisee)
				order by id_stream
--3)
Select titre, pseudo, heure_fin, date_fin_effective,
CASE
    WHEN date_fin_effective > heure_fin
    THEN 'DEPASSEMENT'
    ELSE 'OK'
END as validation,
CASE
    WHEN date_fin_effective > heure_fin
    THEN EXTRACT(EPOCH FROM (date_fin_effective - heure_fin)) / 60
    ELSE 0
END as validation
	from streamer right join creneau on streamer.id_streamer = creneau.id_streamer
		right join stream on stream.id_creneau = creneau.id_creneau
				order by id_stream
--4)
Select COUNT(validation),AVG(depassement) FROM
	(Select titre, pseudo, heure_fin, date_fin_effective,
		CASE
		    WHEN date_fin_effective > heure_fin
		    THEN 'DEPASSEMENT'
		    ELSE 'OK'
		END as validation,
		CASE
		    WHEN date_fin_effective > heure_fin
		    THEN EXTRACT(EPOCH FROM (date_fin_effective - heure_fin)) / 60
		    ELSE 0
		END as depassement
			from streamer right join creneau on streamer.id_streamer = creneau.id_streamer
				right join stream on stream.id_creneau = creneau.id_creneau
						order by id_stream)
	where validation = 'DEPASSEMENT'
		group by validation
